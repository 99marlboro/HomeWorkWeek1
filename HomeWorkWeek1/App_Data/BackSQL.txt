SELECT 
	客戶資料.Id
	,客戶資料.客戶名稱
	,客戶資料.統一編號
	,客戶資料.電話
	,客戶資料.傳真
	,客戶資料.地址
	,客戶資料.Email
	,客戶資料.是否已刪除
	,客戶聯絡人.Id AS 客戶聯絡人Id
	,客戶聯絡人.職稱 AS 客戶聯絡人職稱 
	,客戶聯絡人.姓名 AS 客戶聯絡人姓名
	,客戶聯絡人.Email AS 客戶聯絡人Email
	,客戶聯絡人.手機 AS 客戶聯絡人手機
	,客戶聯絡人.電話 AS 客戶聯絡人電話
	,客戶聯絡人.是否已刪除 AS 聯絡人是否已刪除
	,客戶銀行資訊.Id AS 客戶銀行資訊Id
	,客戶銀行資訊.銀行名稱
	,客戶銀行資訊.銀行代碼
	,客戶銀行資訊.分行代碼
	,客戶銀行資訊.帳戶名稱
	,客戶銀行資訊.帳戶號碼
	,客戶銀行資訊.是否已刪除 AS 銀行資訊是否已刪除
FROM 客戶資料
LEFT JOIN 客戶聯絡人 ON 客戶資料.Id = 客戶聯絡人.客戶Id
LEFT JOIN 客戶銀行資訊 ON 客戶資料.Id = 客戶銀行資訊.客戶Id
